dplyr

dplyr cheatsheet

library(dplyr)

Summarize data

all_data   <- mtcars

summarise(all_data, mean=mean(mpg),
                    sd=sd(mpg))

Group by

summary_df <- all_data %>% 
              group_by(cyl,am) %>%
              summarize(mean=mean(mpg),
                        se=sd(mpg)/sqrt(length(mpg)))

Filter rows

restricted_df <- all_data %>% 
                 filter(gear > 3)

restricted_df <- all_data %>% 
                 filter(gear > 3,
                        am == 1)

all_data <- all_data %>%

Arrange order

arrange(mtcars, mpg)  # ascending (small to large)

arrange(mtcars, desc(mpg)) # descending (large to small)
